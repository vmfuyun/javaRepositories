 4, jrender 的 页面跳转 携带参数   
	使用render-key -----获取对应属性名的值
	render-fun-----对应要 调用的 函数名   
	$(".travels").renderValues({list: dataArr},                         --对应的进行渲染
                            {
                                'getHref': function (item, value) {     /*页面间传数据*/              多参数进行 添加方法名:    函数  item为对应的标签,,,, value 为render-key的值
                                    $(item).attr("href","editTravels.html"+"?id="+value);     		添加属性  然后跳转过去
                                }
                            }
                        );
