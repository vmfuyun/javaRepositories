class StringOperate{
       public long doWork(){
        long begin=System.currentTimeMills();//开始时间
        for(int i=0;i<10000;i++){
            str+=i;
        }
        long end=System.currentTimeMillis();//结束时间
        long time=end-begin;
        return time;
    }
}
